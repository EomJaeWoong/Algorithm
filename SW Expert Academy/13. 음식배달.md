# Algorithm

### 음식 배달

> 문제 링크 : 없음



###### 나의 풀이

1. 집, 스토어의 정보를 입력받음

2. 스토어를 선택하는 부분집합을 구하여 거리와 운영비를 합해본다

3. 합의 최소값을 취한다

   

###### 풀이

- ArrayList로 풀이

~~~java
package com.algorithm.problem;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.StringTokenizer;

/*
 * 10888. 음식배달
 * JAVA_서울_6반_엄재웅
 */

public class sw10888_FoodDelivery_엄재웅 {
	private static boolean[] selected;
	private static ArrayList<int[]> home, store;
	private static int min;
	
	public static void main(String[] args) throws Exception {
		BufferedReader br  = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		int T = Integer.parseInt(br.readLine());
		
		for (int t = 1; t <= T; t++) {
			sb.append('#').append(t).append(' ');
			int N = Integer.parseInt(br.readLine());
			home = new ArrayList<int[]>();
			store = new ArrayList<int[]>();
			min = Integer.MAX_VALUE;
			
			// 집, 스토어 입력받기
			for (int y = 0; y < N; y++) {
				StringTokenizer st = new StringTokenizer(br.readLine());
				for (int x = 0; x < N; x++) {
					int p = Integer.parseInt(st.nextToken());
					if(p == 1) {
						home.add(new int[] {x, y});
					} else if(p > 1) {
						store.add(new int[] {x, y, p});
					}
				}
			}
			
			selected = new boolean[store.size()];

			// 부분 집합 구하기
			subset(0);
			
			sb.append(min).append('\n');
		}	// end test case
		
		System.out.println(sb);
	}
	
	private static void subset(int cnt) {
		if(cnt == store.size()) {
			int sum = 0, is = 0;
			
			// 먼저 선택된 상점의 운영비를 더함
			for (int s = 0; s < store.size(); s++) {
				if(selected[s]) {
					sum += store.get(s)[2];
					is++;
				}
			}
			
			// 공집합이면 그냥 빠져나감
			if(0 == is)	return;
			
			// 집마다 거리를 잼
			for (int h = 0; h < home.size(); h++) {
				int mindis = Integer.MAX_VALUE;
				// 선택된 상점과 집의 거리 최소값을 구함
				for (int s = 0; s < store.size(); s++) {
					if(selected[s]) {
						int distance = Math.abs(home.get(h)[0] - store.get(s)[0]) + 
								Math.abs(home.get(h)[1] - store.get(s)[1]);
						
						mindis = Math.min(mindis, distance);
					}
				}
				
				sum += mindis;
			}
				
			
			min = Math.min(min, sum);
			return;
		}
		
		// 다음 부분집합
		selected[cnt] = true;
		subset(cnt + 1);
		selected[cnt] = false;
		subset(cnt + 1);
	}
}
~~~

~~~java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

/*
 * 10888. 음식배달
 * JAVA_서울_6반_엄재웅
 */

public class Solution {
	private static boolean[] selected;
	private static int[][] home, store;
	private static int idxh, idxs;
	private static int min;
	
	public static void main(String[] args) throws Exception {
		BufferedReader br  = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		int T = Integer.parseInt(br.readLine());
		
		for (int t = 1; t <= T; t++) {
			sb.append('#').append(t).append(' ');
			int N = Integer.parseInt(br.readLine());
			home = new int[100][];
			store = new int[10][];
			idxh = 0;
			idxs = 0;
			min = Integer.MAX_VALUE;
			
			for (int y = 0; y < N; y++) {
				StringTokenizer st = new StringTokenizer(br.readLine());
				for (int x = 0; x < N; x++) {
					int p = Integer.parseInt(st.nextToken());
					if(p == 1) {
						home[idxh++] = new int[] {x, y};
					} else if(p > 1) {
						store[idxs++] = new int[] {x, y, p};
					}
				}
			}
			
			selected = new boolean[idxs];

			subset(0);
			
			sb.append(min).append('\n');
		}	// end test case
		
		System.out.println(sb);
	}
	
	private static void subset(int cnt) {
		if(cnt == idxs) {
			int sum = 0, is = 0;
			
			for (int s = 0; s < idxs; s++) {
				if(selected[s]) {
					sum += store[s][2];
					is++;
				}
			}
			
			if(0 == is)	return;
			
			for (int h = 0; h < idxh; h++) {
				int mindis = Integer.MAX_VALUE;
				for (int s = 0; s < idxs; s++) {
					if(selected[s]) {
						int distance = Math.abs(home[h][0] - store[s][0]) + 
								Math.abs(home[h][1] - store[s][1]);
						
						mindis = Math.min(mindis, distance);
					}
				}
				
				sum += mindis;
			}
				
			
			min = Math.min(min, sum);
			return;
		}
		
		selected[cnt] = true;
		subset(cnt + 1);
		selected[cnt] = false;
		subset(cnt + 1);
	}
}
~~~
