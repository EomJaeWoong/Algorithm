# Algorithm

### 딱지 치기

> 문제 링크 : https://www.acmicpc.net/problem/14696



###### 나의 풀이

1. 라운드마다 두 딱지를 입력받고

2. 딱지 수를 내림차순으로 정렬함

3. 탐색하면서 차이가 나는 값이 있으면 큰값이 승자

4. 모든 탐색을 했으면 길이가 더 긴사람이 승자

5. 그 외에는 무승부




###### 풀이

~~~java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.Collections;
import java.util.StringTokenizer;

/*
 * 14696. 딱지치기
 * JAVA_서울_6반_엄재웅
 */

public class Solution {
	// 별, 동그라미, 네모, 세모 1,2,3,4
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		StringTokenizer st;
		int N = Integer.parseInt(br.readLine());
		
		for (int n = 0; n < N; n++) {
			char result = 'D';
			
			// A
			st = new StringTokenizer(br.readLine(), " ");
			int A = Integer.parseInt(st.nextToken());
			Integer[] a = new Integer[A];
			for (int i = 0; i < A; i++) {
				a[i] = Integer.parseInt(st.nextToken());
			}
			
			// B
			st = new StringTokenizer(br.readLine(), " ");
			int B = Integer.parseInt(st.nextToken());
			Integer[] b = new Integer[B];
			for (int i = 0; i < B; i++) {
				b[i] = Integer.parseInt(st.nextToken());
			}
			
			// 정렬
			Arrays.sort(a, Collections.reverseOrder());
			Arrays.sort(b, Collections.reverseOrder());
			
			// 비교
			for (int i = 0; i < (A > B? B : A); i++) {
				if(a[i] > b[i] || i == B) {			// A 승리
					result = 'A';
					break;
				} else if(a[i] < b[i] || i == A) {	// B 승리
					result = 'B';
					break;
				}
			}

			// 검사한 수가 같으므로 길이로 비교
			if(result == 'D')
				result = A == B? 'D' : A > B? 'A': 'B';
				
			sb.append(result).append('\n');
		}
		
		System.out.println(sb);
	}
}
~~~
