# Algorithm

### Z

> 문제 링크 : https://www.acmicpc.net/problem/1074

> 재귀



###### 풀이

~~~java
import java.util.Scanner;

/*
 * 1074. Z
 * JAVA_서울_6반_엄재웅
 */

/*
 * TODO
 * 1. N, Z, r, c 입력받기
 * 2. 배열에 넣을 cnt 선언
 * 3. 배열 순회를 위한 재귀함수 선언
 *   1. 기저조건 n = 1이면 cnt값을 넣고 + 1
 *   2. n = 1이 아니면 Z형식에 따라 N/2사이즈로 4번 재귀를 호출한다
 */

public class bj1074_Z_엄재웅 {
	private static int N;
	// private static int[][] Z;
	private static int r, c;
	private static int cnt, answer;

	public static void main(String[] args) throws Exception {
		Scanner sc = new Scanner(System.in);

		N = (int) Math.pow(2, sc.nextInt());
		// Z = new int[N][N];
		r = sc.nextInt();
		c = sc.nextInt();
		cnt = 0;

		travelZ(N, 0, 0);

		System.out.println(answer);
	}

	private static void travelZ(int n, int x, int y) {
		int divide = n / 2;

		if (x == r && y == c) {
			answer = cnt;
		}
		if (n <= 1) {
			cnt++;
			return;
		}

		// 나눠진 4개를 다시 재귀로 들어감
		travelZ(divide, x, y);
		travelZ(divide, x, y + divide);
		travelZ(divide, x + divide, y);
		travelZ(divide, x + divide, y + divide);

	}
}
~~~
