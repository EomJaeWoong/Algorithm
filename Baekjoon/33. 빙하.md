# Algorithm

### 빙하

> 문제 링크 : https://www.acmicpc.net/problem/2573



* 문제를 잘 읽읍시다(결과가 0이 되는 경우를 체크하지 않음)

  

###### 나의 풀이

1. 빙하가 두덩이 이상으로 갈라졌는지는 dfs로 체크한다

   만약 빙하덩이가 없으면 0으로 출력

2. 주위 바다면의 0 갯수를 계산한다(배열로)

3. 위에 배열을 이용하여 원래 빙하에서 빼준다

4. +1년




###### 풀이

~~~java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

/*
 * 2573. 빙산
 * JAVA_서울_6반_엄재웅
 */

public class Main {
	private static int N, M, result;
	private static int[][] ground, dir = {
			{-1, 0}, {1, 0}, {0, -1}, {0, 1}
	};	// 4방향
	private static boolean[][] visited;
	
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine(), " ");
		
		// N, M 입력
		N = Integer.parseInt(st.nextToken());
		M = Integer.parseInt(st.nextToken());
		
		// 빙산 정보 입력
		ground = new int[N][M];
		for (int y = 0; y < N; y++) {
			st = new StringTokenizer(br.readLine(), " ");
			for (int x = 0; x < M; x++) {
				ground[y][x] = Integer.parseInt(st.nextToken());
			}
		}

		calAround();
		
		System.out.println(result);
	}

	private static void calAround() {
		int[][] minus = new int[N][M];
		
		while(true) {
			boolean isExist = false;
			visited = new boolean[N][M];
			
			// 빙산이 한덩이인지 아닌지 체크
			for (int y = 0; y < N; y++) {
				for (int x = 0; x < M; x++) {
					if(ground[y][x] != 0 && !visited[y][x]) {
						if(isExist)	return;
						
						dfs(x, y);
						isExist = true;
					}
				}
			}
			
			// 체크되지 않으면 빙하가 전부 녹은 것임
			if(!isExist) {
				result = 0;
				return;
			}
			
			minus = new int[N][M];
			
			// 주위 바다 면 계산
			for (int y = 0; y < N; y++) {
				for (int x = 0; x < M; x++) {
					if(ground[y][x] != 0) {
						for (int d = 0; d < 4; d++) {
							int ax = x + dir[d][0];
							int ay = y + dir[d][1];
							
							if(isIn(ax, ay) && ground[ay][ax] == 0) 
								minus[y][x]++;
						}
					}
				}
			}
			
			// 빼줌
			for (int y = 0; y < N; y++) {
				for (int x = 0; x < M; x++) {
				ground[y][x] = ground[y][x] - minus[y][x] > 0? 
						ground[y][x] - minus[y][x] : 0;
				}
			}
			
			result++;
		}
	}

	private static void dfs(int x, int y) {
		visited[y][x] = true;
		
		for (int d = 0; d < 4; d++) {
			int nx = x + dir[d][0];
			int ny = y + dir[d][1];
			
			if(isIn(nx, ny) && ground[ny][nx] != 0 && !visited[ny][nx])
				dfs(nx, ny);
		}
	}

	private static boolean isIn(int x, int y) {
		return x < 0 || y < 0 || x >= M || y >= N? false : true;
	}
}
~~~
